服务提供商 providers


1. 在整个应用中提供服务
    import { Injectable } from '@angular/core';
      @Injectable({
      providedIn: 'root',
      })
      export class UserService {
      }
2. 在模块中提供服务
    @Injectable({
      // 首选方式 
       providedIn: UserModule,   // 可以摇树优化
      })

或

    import { UserService } from './user.service';
      @NgModule({
        providers: [ UserService ], // 无法摇树优化
      })
      export class UserModule {}
3. 在组件中提供服务
    @Component({
        providers: [UserService]
      });
