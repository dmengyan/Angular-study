路由定义配置


1. 生成路由模块,并配置到app模块

ng g module app-routing --module=app


2. 配置路由模块

import { Routes, RouterModule } from '@angular/router';

@NgModule ({ imports: [ RouterModule.forRoot ( appRoutes ) ], exports: [RouterModule] } )


2. 定义预加载的组件（引入该组件）

import { XxComponent } from './xx/xx.component';

const routes: Routes = [ { path: 'xx', component: XxComponent } ];


2. 定义惰性加载的模块（不引入）

const routes: Routes = [{
path: 'xx', loadChildren: './xx/xx.module#XxModule'
}];


3. 使用

< router-outlet >< /router-outlet >

[routerLink]="['/a']"
